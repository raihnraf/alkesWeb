      let valid = true;
      const required = this.querySelectorAll('[required]');
      required.forEach(field => {
        const errorEl = field.parentElement.querySelector('.form-error');
        if (errorEl) errorEl.remove();
        field.classList.remove('error');
        if (!field.value.trim()) {
          valid = false;
          field.classList.add('error');
          const err = document.createElement('p');
          err.className = 'form-error';
          err.textContent = 'Mohon isi bidang ini dengan benar.';
          field.parentElement.appendChild(err);
        }
      });
      if (valid) {
        const toast = document.createElement('div');
        toast.className = 'form-success';
        toast.textContent = 'Pesan Anda telah tercatat. Tim kami akan menghubungi Anda segera.';
        document.body.appendChild(toast);
        setTimeout(() => toast.remove(), 4000);
        this.reset();

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Form input → Client-side JS | User-entered data processed only in browser, never transmitted (no backend) |
| External links (wa.me, mailto, tel) | Links open in user's apps — no data interception risk |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-03 | Spoofing | WhatsApp deep link | accept | Uses official wa.me domain with HTTPS — no spoofing risk from our side |
| T-05-04 | Tampering | Form validation JS | mitigate | Client-side validation is display-only; no data persistence means no tampering impact. Validation prevents empty submissions from showing false success. |
| T-05-05 | Information Disclosure | Form fields | accept | No backend — form data never leaves the browser. Success toast confirms display-only behavior. |
| T-05-06 | Repudiation | Form submission | accept | Display-only form with no audit trail. Users confirmed via toast; no repudiation risk for static demo. |
</threat_model>
